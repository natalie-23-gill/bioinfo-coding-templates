mpg <- filter(mpg, manufacturer %in% c(
  "ford", "honda", "dodge", "toyota",
  "nissan", "pontiac", "subaru"
))
mpg |>
  ggplot(aes(y = hwy, x = manufacturer, fill = manufacturer)) +
  geom_violin() +
  geom_boxplot(width = .1, fill = "white") +
  labs(title = "Highway MPG")

     linetype = "dashed",
